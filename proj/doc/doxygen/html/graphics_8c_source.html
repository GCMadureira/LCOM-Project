57int (vg_set_VBE_mode)(uint16_t mode){
58 reg86_t r86;
59 memset(&r86, 0, sizeof(r86));
60
61 r86.intno = VIDEO_INT_INSTRUCTION;
62 r86.ah = VBE_FUNCTION;
63 r86.al = VBE_SET_MODE;
64 r86.bx = LINEAR_FRAME_BUFFER | mode;
65
66 if(sys_int86(&r86) != OK) return 1;
67 if(r86.al != VBE_FUNCTION_SUPPORTED) return 1; // function not supported
68 if(r86.ah) return 1; // function not successful
69
70 return 0;
71}

74int (vg_map_video_memory)(uint16_t mode){
75 struct minix_mem_range mr;
76 unsigned int vram_base;
77 unsigned int vram_size;
78
79 int r;
80
81 /* Use VBE function 0x01 to initialize vram_base and vram_size */
82
83 // get the mode information
84 if(vbe_get_mode_info(mode, &mode_info)) return 1;
85
86 // ceil accounts for bits per pixel like mode 0x110 (not multiple of 8)
87 bytes_per_pixel = ceil((double)vg_get_bits_per_pixel()/8);
88
89 vram_base = mode_info.PhysBasePtr;
90 vram_size = vg_get_hres()*vg_get_vres()*bytes_per_pixel;
91
92 /* Allow memory mapping */
93
94 mr.mr_base = (phys_bytes) vram_base;
95 mr.mr_limit = mr.mr_base + vram_size;
96
97 if(OK != (r = sys_privctl(SELF, SYS_PRIV_ADD_MEM, &mr))){
98 panic("sys_privctl (ADD_MEM) failed: %d\n", r);
99 return 1;
100 }
101
102 /* Map memory */
103
104 buffer = (char*)malloc(vram_size);
105 video_mem = vm_map_phys(SELF, (void *)mr.mr_base, vram_size);
106
107 if(video_mem == MAP_FAILED) {
108 panic("couldn't map video memory");
109 return 1;
110 }
111
112 return 0;
113}

150int (vg_draw_rectangle32)(int32_t x, int32_t y, uint16_t width, uint16_t height, uint32_t color) {
151 if(buffer == NULL) return 1;
152 if(x >= (int32_t)vg_get_hres() || x + (int32_t)width < 0 ||
153 y >= (int32_t)vg_get_vres() || y + (int32_t)height < 0 ) return 1;
154
155 if(color == COLOR32_TRANSPARENT) return 0; // no need to even draw
156
157 uint32_t x_start = MAX(0, x), x_end = MIN((int32_t)vg_get_hres(), x + (int32_t)width);
158 uint32_t y_start = MAX(0, y), y_end = MIN((int32_t)vg_get_vres(), y + (int32_t)height);
159
160 uint32_t* vmem_address = (uint32_t *)buffer + y_start * vg_get_hres();
161
162 for(uint32_t y_off = y_start; y_off < y_end; ++y_off){
163 for(uint32_t x_off = x_start; x_off < x_end; ++x_off){
164 memcpy(vmem_address + x_off, &color, 4);
165 }
166 vmem_address += vg_get_hres(); // move to next vmem line
167 }
168
169 return 0;
170}

177 uint32_t x_start = MAX(0, x), x_end = MIN((int32_t)vg_get_hres(), x + (int32_t)img_info->width);
178 uint32_t y_start = MAX(0, y), y_end = MIN((int32_t)vg_get_vres(), y + (int32_t)img_info->height);
179
180 uint32_t* vmem_address = (uint32_t *)buffer + y_start*vg_get_hres();
181 uint32_t* img_address = (uint32_t *)img_info->bytes + (y_start - y)*img_info->width;
182
183 for(uint32_t y_off = y_start; y_off < y_end; ++y_off){
184 for(uint32_t x_off = x_start; x_off < x_end; ++x_off){
185 if(*(img_address + x_off - x) != COLOR32_TRANSPARENT)
186 memcpy(vmem_address + x_off, img_address + x_off - x, 4);
187 }
188 vmem_address += vg_get_hres(); // move to next vmem line
189 img_address += img_info->width; // move to next image line
190 }
191
192 return 0;
193}
