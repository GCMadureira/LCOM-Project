188static bool (check_heart_collision)(entity* heart, entity* player) {
189 // Simple box collision detection
190 return (heart->pos_x < player->pos_x + player->animations[0].sprites[0]->width &&
191 heart->pos_x + heart->animations[0].sprites[0]->width > player->pos_x &&
192 heart->pos_y < player->pos_y + player->animations[0].sprites[0]->height &&
193 heart->pos_y + heart->animations[0].sprites[0]->height > player->pos_y);
194}

196static void (handle_heart_collisions)(arena* arena) {
197 entity_node* current_heart = arena->hearts;
198 entity_node* previous_heart = NULL;
199
200 while(current_heart != NULL) {
201 entity* heart = current_heart->entity;
202
203 // Check for collision with player
204 if (check_heart_collision(heart, arena->player)) {
205 // Increase player health if not at max
206 if (arena->player->health < 4) {
207 arena->player->health++;
208 }
209
210 // Remove the heart
211 entity_node* victim = current_heart;
212 current_heart = current_heart->next_entity;
213 if (previous_heart == NULL)
214 arena->hearts = current_heart;
215 else
216 previous_heart->next_entity = current_heart;
217 entity_node_destroy(victim);
218 continue;
219 }
220
221 previous_heart = current_heart;
222 current_heart = current_heart->next_entity;
223 }
224}

263int (arena_process_frame)(arena* arena) {
264 // If handle_enemy_collisions() returns true, the player has collided with an enemy
265 // Add a damage cooldown of 60 frames (1 second at 60 FPS)
266 if (handle_enemy_collisions(arena) && (get_current_frame() - last_damage_time >= 60)) {
267 arena->player->health--;
268 if (arena->player->health > 0)
269 last_damage_time = get_current_frame();
270 else
271 return 1;
272 }
273 // handle heart collisions
274 handle_heart_collisions(arena);
275
276 // spawn the enemies and hearts
277 handle_enemy_spawning(arena);
278 handle_heart_spawning(arena);
279 handle_secret_spawning(arena);
280 handle_auto_attack(arena);
281
282 // move everything
283 arena_move_entities(arena);
284
285 // update the active time
286 if ((get_current_frame() - arena_starting_frame) % 60 == 0)
287 ++arena_game_time;
288
289 return 0;
290}
