82static int (process_menu)(){
83 input_event event;
84 while(events_get_next(&event) == 0) {
85 if(event.event_type == KEYBOARD_EVENT) {
86 switch (event.scancode_byte1) {
87 case KEY_MK_W:
88 menu_select_option_up(active_menu);
89 break;
90 case KEY_MK_S:
91 menu_select_option_down(active_menu);
92 break;
93 case KEY_MK_ENTER: {
94 // main menu processing
95 if(game_state == MAIN_MENU) {
96 if (active_menu->menu_status == 0) {
97 game_state = GAME;
98 active_arena = arena_create();
99 setup_arena_controller(); // setup the controllers for a new game
100 setup_enemy_controller();
101 menu_destroy(active_menu);
102 active_menu = NULL;
103 return 0;
104 }
105 else if (active_menu->menu_status == 1) game_state = QUIT;
106 }
107 else if(game_state == GAME_OVER_MENU) {
108 if (active_menu->menu_status == 0) {
109 game_state = GAME;
110 if(active_arena != NULL) arena_destroy(active_arena);
111 active_arena = arena_create();
112 setup_arena_controller(); // setup the controllers for a new game
113 setup_enemy_controller();
114 menu_destroy(active_menu);
115 active_menu = NULL;
116 return 0;
117 }
118 else if (active_menu->menu_status == 1) {
119 game_state = MAIN_MENU;
120 if(active_arena != NULL) arena_destroy(active_arena);
121 active_arena = NULL;
122 menu_destroy(active_menu);
123 active_menu = menu_create_main();
124 return 0;
125 }
126 }
127 break;
128 }
129 }
130 }
131 }
132
133 return 0;
134}

137static int (process_game)(){
138 entity* player = active_arena->player;
139 entity* mouse = active_arena->mouse;
140 input_event event;
141
142 while(events_get_next(&event) == 0) {
143 if(event.event_type == KEYBOARD_EVENT) {
144 switch (event.scancode_byte1) {
145 case KEY_MK_W: //moving up
146 player->speed_y = -3;
147 break;
148 case KEY_MK_S: //moving down
149 player->speed_y = 3;
150 break;
151 case KEY_MK_D: //moving right
152 player->last_horizontal_direction = true;
153 player->speed_x = 3;
154 break;
155 case KEY_MK_A: //moving left
156 player->last_horizontal_direction = false;
157 player->speed_x = -3;
158 break;
159 case KEY_BK_W:
160 if(player->speed_y < 0) player->speed_y = 0;
161 break;
162 case KEY_BK_S:
163 if(player->speed_y > 0) player->speed_y = 0;
164 break;
165 case KEY_BK_D:
166 if(player->speed_x > 0) player->speed_x = 0;
167 break;
168 case KEY_BK_A:
169 if(player->speed_x < 0) player->speed_x = 0;
170 break;
171 }
172 }
173 else if(event.event_type == MOUSE_EVENT) {
174 mouse->pos_x += (double)event.mouse_packet.delta_x/2;
175 mouse->pos_y += -(double)event.mouse_packet.delta_y/2;
176
177 // bound the position inside the screen
178 mouse->pos_x = MIN(MAX(0, mouse->pos_x), vg_get_hres() - cursor_img.width);
179 mouse->pos_y = MIN(MAX(0, mouse->pos_y), vg_get_vres() - cursor_img.height);
180
181 if(event.mouse_packet.lb)
182 handle_ranged_attack(active_arena);
183 }
184 }
185
186 return 0;
187}

189int (process_frame)() {
190 // The game state can change on the process part, need to separate it
191 if(game_state == MAIN_MENU || game_state == GAME_OVER_MENU) process_menu();
192 else if(game_state == GAME) process_game();
193
194
195 if(game_state == MAIN_MENU) draw_menu(active_menu);
196 else if(game_state == GAME) {
197 // If arena_process_frame == 1 it means check_collisions == 1
198 if(arena_process_frame(active_arena) == 1) {
199 // If player Health == 0 return to menu
200 // We are doing the entire "changing health bar sprite" on the viewers
201 if (active_arena->player->health == 0) {
202 update_high_score(get_arena_game_time());
203 game_state = GAME_OVER_MENU;
204 active_menu = menu_create_game_over();
205 }
206 }
207 else draw_arena(active_arena);
208 }
209 else if(game_state == GAME_OVER_MENU) {
210 draw_arena(active_arena);
211 draw_menu(active_menu);
212 }
213
214 vg_show_frame();
215 ++frame;
216
217 return 0;
218}
