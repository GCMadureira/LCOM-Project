60int (timer_get_conf)(uint8_t timer, uint8_t *st) {
61 switch (timer){

63 case TIMER_0:

65 sys_outb(TIMER_CTRL, TIMER_RB_CMD | TIMER_RB_SEL(0)); //0b11000010

84int (timer_display_conf)(uint8_t timer, uint8_t st, enum timer_status_field field) {
85 union timer_status_field_val conf;
86 switch (field) {
87 case tsf_all:
88 // Display entire status byte
89 conf.byte = st;
90 break;
91 case tsf_initial: {
92 // Display initialization mode | Located on bits 4-5
93 uint8_t initValue = (st >> 4) & 0x03;
94 if(initValue == 0) conf.in_mode = INVAL_val;
95 else if(initValue == 1) conf.in_mode = LSB_only;
96 else if(initValue == 2) conf.in_mode = MSB_only;
97 else if(initValue == 3) conf.in_mode = MSB_after_LSB;
98 break;
99 }
100 case tsf_mode: {
101 // Display the counting mode | Located on bits 1-3
102 uint8_t count_mode = (st >> 1) & 0x07;
103 // since the third bit does not matter, 0b110 also means mode 2 and 0b111 means mode 3
104 if(count_mode == 6) conf.count_mode = 2;
105 else if(count_mode == 7) conf.count_mode = 3;
106 else conf.count_mode = (st >> 1) & 0x07;
107 break;
108 }
109 case tsf_base:
110 // Display counting base | Located on bit 0 | Either "Binary" or "BCD"
111 conf.bcd = st & 0x01;
112 break;
113 default:
114 return 1; // Invalid, return error
115 }
116 return timer_print_config(timer, field, conf);
117}
