23bool (kbc_valid_output)(bool aux){
24 uint8_t status;

26 if(status & (KBC_PAR_ERR | KBC_TO_ERR)) return false; //parity error or timeout error
27 if(!(status & KBC_ST_OBF)) return false; //output buffer empty
28
29 if(aux) return status & KBC_ST_AUX; //mouse data
