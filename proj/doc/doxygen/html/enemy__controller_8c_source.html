123bool (handle_enemy_collisions)(arena* arena) {
124 entity_node* current_enemy = arena->enemies;
125 entity_node* previous_enemy = NULL;
126
127 outer:
128 while(current_enemy != NULL) {
129 entity* enemy = current_enemy->entity;
130 attack_node* current_attack = arena->player_attacks;
131
132 //check collisions with the attacks of the player, remove the enemy if health = 0
133 while(current_attack != NULL) {
134 attack* attack = current_attack->attack;
135 if(check_attack_collision(enemy, attack)) {
136 if(enemy->health <= attack->damage) {
137 // Remove the enemy
138 entity_node* victim = current_enemy;
139 current_enemy = current_enemy->next_entity;
140 if(previous_enemy == NULL)
141 arena->enemies = current_enemy;
142 else
143 previous_enemy->next_entity = current_enemy;
144 entity_node_destroy(victim);
145 --enemy_count;
146 goto outer;
147 }
148 else current_enemy->entity->health -= current_attack->attack->damage;
149 }
150 current_attack = current_attack->next_attack;
151 }
152
153 // Check for collision with player
154 if (check_player_collision(enemy, arena->player)) {
155 return true; // Collision detected
156 }
157
158 previous_enemy = current_enemy;
159 current_enemy = current_enemy->next_entity;
160 }
161
162 return false; // No collision
163}
