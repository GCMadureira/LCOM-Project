50void (entity_move)(entity* entity){
51 // dividing by the lenght normalizes the vector, then multiply it by the speed
52 double lenght = sqrt(entity->speed_x * entity->speed_x + entity->speed_y * entity->speed_y);
53 if(lenght == 0) return;
54 double scale = entity->speed_multiplier/lenght;
55
56 entity->pos_x += scale * (entity->speed_x);
57 entity->pos_y += scale * (entity->speed_y);
58
59 // bound the position inside the background image size (arena size)
60 entity->pos_x = MIN(MAX(0, entity->pos_x), game_background_img.width - entity->animations[0].sprites[0]->width);
61 entity->pos_y = MIN(MAX(0, entity->pos_y), game_background_img.height - entity->animations[0].sprites[0]->height);
62}
