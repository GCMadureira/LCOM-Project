23int (game_init)() {
24 uint8_t timer_int_no = 0, keyboard_int_no = 1, mouse_int_no = 2;
25 const int frame_rate = 60;
26
27 if(timer_set_frequency(TIMER_0, frame_rate)) return 1; // set the frame rate on timer 0
28 if(timer_subscribe_int(&timer_int_no)) return 1; // subscribe the timer 0 interruptions
29 if(keyboard_subscribe_int_exclusive(&keyboard_int_no)) return 1; // subscribe keyboard ints
30 if(mouse_stream_enable_data_reporting()) return 1; // enable reporting on the mouse
31 if(mouse_subscribe_int_exclusive(&mouse_int_no)) return 1; // subscribe the interruptions reported
32 if(vg_graphics_init(0x14C)) return 1; // set the graphics mode as 0x14C and map the vmem
33
34 if(setup_game()) return 1;
35
36 return 0;
37}

64int (proj_main_loop)() {
65 if(game_init()) {
66 printf("Error: could not initialize the game\n");
67 if(game_exit()) printf("Warning: something went wrong while cleaning up\n");
68 return 1;
69 }
70
71 int ipc_status, r;
72 message msg;
73
74 // main loop
75 while(keyboard_get_scancode() != ESC_KEY_BREAKCODE && get_game_state() != QUIT) {
76 if ( (r = driver_receive(ANY, &msg, &ipc_status)) != 0 ) {
77 printf("driver_receive failed with: %d", r);
78 continue;
79 }
80 if (is_ipc_notify(ipc_status)) {
81 switch (_ENDPOINT_P(msg.m_source)) {
82 case HARDWARE:
83 if (msg.m_notify.interrupts & BIT(0)) { // timer, execute one frame
84 process_frame();
85 }
86 if (msg.m_notify.interrupts & BIT(1)) { // keyboard
87 event_handle_keyboard();
88 }
89 if (msg.m_notify.interrupts & BIT(2)) { // mouse
90 event_handle_mouse();
91 }
92 break;
93 }
94 }
95 }
96
97 if(game_exit()) {
98 printf("Warning: something went wrong while cleaning up\n");
99 return 1;
100 }
101 return 0;
102}
